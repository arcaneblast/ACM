ll sum(ll a,ll b){
    ll res1=((a*(a+1))/2);
    ll res2=((b*(b+1))/2);
    return (res2-res1)%moder;
}
ll func(ll n){
    ll limit=floor(sqrt(n));
    ll res=0;
    for(int i=1;i<=limit;i++){
        ll hi=(n/i);
        ll lo=max(n/(i+1),limit);
        res=(res+i*sum(lo,hi)%moder)%moder;
        res=(res+(i*(n/i))%moder)%moder;
    }
    return res;
}

    int cases;
    scanf("%d",&cases);
    while(cases--){
        ll n;
        scanf("%lld",&n);
        printf("%lld\n",func(n));
    }
}
